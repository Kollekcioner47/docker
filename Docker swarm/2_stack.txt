опишем наш стэк
создадим файл stack.yml на управляющей ноде
в нём пропишем:

version: "3"

services:
  wordpress:
    image: wordpress
    ports:
      - 81:80
    environment:
      - WORDPRESS_DB_HOST=mysql2
      - WORDPRESS_DB_USER=wp
      - WORDPRESS_DB_PASSWORD=wp_pass
      - WORDPRESS_DB_NAME=wp
    deploy:
      replicas: 3
      restart_policy:
        condition: on-failure
  
  mysql2:
    image: mysql:8.0
    environment:
      - MYSQL_ROOT_PASSWORD=root
      - MYSQL_USER=wp
      - MYSQL_PASSWORD=wp_pass
      - MYSQL_DATABASE=wp
    deploy:
      placement::
        constraint: [node.role == manager]
  
  phpmyadmin:
    image: phpmyadmin
    ports:
      - 8080:80
    environment:
      - PMA_HOST=mysql2
    deploy:
      replicas: 2


сохраним файл и запустим его
docker stack deploy -c stack.yml wp


посмотреть список стэков
docker stack ls

какие сервисы запущены на стеке

docker stack services pw
docker service ls

список контейнеров 
docker service ps pw ps_mysql.1

инспекция сервиса
docker service inpsect pw_wordpress
